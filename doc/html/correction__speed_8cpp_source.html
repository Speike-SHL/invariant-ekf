42int main() {
43 // Initialize filter
44 InEKF filter;

48 ifstream infile("./src/data/correction_speed_test_data.txt");
49 string line;
50 Eigen::Matrix<double,6,1> m, m_last;
51 double t, t_last;
52 m_last << 0,0,0,0,0,0;
53 t_last = 0;
54 vectorPairIntVector6d measurements_vec;
55 vectorLandmarks measured_landmarks;
56
57 // Loop through data file and read in measurements line by line
58 while (getline(infile, line)){
59 vector<string> measurement;
60 boost::split(measurement,line,boost::is_any_of(" "));
61 // Handle measurements
62 if (measurement[0].compare("IMU")==0){
63 t = stod98(measurement[1]);
64 m << stod98(measurement[2]),
65 stod98(measurement[3]),
66 stod98(measurement[4]),
67 stod98(measurement[5]),
68 stod98(measurement[6]),
69 stod98(measurement[7]);
70 measurements_vec.push_back(pair<double,Eigen::Matrix<double,6,1> > (t, m));
71
72 }
73 else if (measurement[0].compare("LANDMARK")==0){
74 t = stod98(measurement[1]);
75 for (int i=2; i<measurement.size(); i+=4) {
76 int id = stod98(measurement[i]);
77 Eigen::Vector3d p_bl;
78 p_bl << stod98(measurement[i+1]),
79 stod98(measurement[i+2]),
80 stod98(measurement[i+3]);
81 Landmark landmark(id, p_bl);
82 measured_landmarks.push_back(landmark);
83 }
84 }
85 }
86
87 // Propagate all IMU data
88 cout << "Propagating " << measurements_vec.size() << " IMU measurements...\n";
89 for (vectorPairIntVector6dIterator it=measurements_vec.begin(); it!=measurements_vec.end(); ++it) {
90 // Propagate using IMU data
91 t = it->first;
92 m = it->second;
93 double dt = t - t_last;
94 if (dt > DT_MIN && dt < DT_MAX) {
95 filter.Propagate(m_last, dt);
96 }
97 // Store previous timestamp
98 t_last = t;
99 m_last = m;
100 }
101 cout << filter.getState() << endl;
102
103 // Correct using landmark data
104 cout << "Correcting " << measured_landmarks.size() << " landmark measurements...\n";
105 int64_t max_duration = 0;
106 int64_t sum_duration = 0;
107 for (vectorLandmarksIterator it=measured_landmarks.begin(); it!=measured_landmarks.end(); ++it) {
108 vectorLandmarks landmarks;
109 landmarks.push_back(*it);
110 ptime start_time = second_clock::local_time();
111 filter.CorrectLandmarks(landmarks);
112 //cout << filter.getState() << endl;
113 ptime end_time = second_clock::local_time();
114 int64_t duration = (end_time - start_time).total_nanoseconds();
115 //cout << "duration: " << duration << endl;
116 sum_duration += duration;
117 if (duration > max_duration)
118 max_duration = duration;
119 }
120 cout << filter.getState() << endl;
121 cout << "max duration: " << max_duration << endl;
122 cout << "average duration: " << double(sum_duration)/(measured_landmarks.size()/3) << endl;
123 return 0;
124}
