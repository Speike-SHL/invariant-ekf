34using namespace inekf;

36int main() {
37 // ---- Initialize invariant extended Kalman filter ----- //
38 RobotState initial_state;
39
40 // Initialize state mean
41 Eigen::Matrix3d R0;
42 Eigen::Vector3d v0, p0, bg0, ba0;
43 R0 << 1, 0, 0, // initial orientation
44 0, -1, 0, // IMU frame is rotated 90deg about the x-axis
45 0, 0, -1;
46 v0 << 0,0,0; // initial velocity
47 p0 << 0,0,0; // initial position
48 bg0 << 0,0,0; // initial gyroscope bias
49 ba0 << 0,0,0; // initial accelerometer bias
50 initial_state.setRotation(R0);
51 initial_state.setVelocity(v0);
52 initial_state.setPosition(p0);
53 initial_state.setGyroscopeBias(bg0);
54 initial_state.setAccelerometerBias(ba0);
55
56 // Initialize state covariance
57 NoiseParams noise_params;
58 noise_params.setGyroscopeNoise(0.01);
59 noise_params.setAccelerometerNoise(0.1);
60 noise_params.setGyroscopeBiasNoise(0.00001);
61 noise_params.setAccelerometerBiasNoise(0.0001);
62 noise_params.setLandmarkNoise(0.1);
63
64 // Initialize filter
65 InEKF filter(initial_state, noise_params);
66 cout << "Noise parameters are initialized to: \n";
67 cout << filter.getNoiseParams() << endl;
68 cout << "Robot's state is initialized to: \n";
69 cout << filter.getState() << endl;
70
71 // --- Optionally initialize prior landmarks --- //
72 mapIntVector3d prior_landmarks;
73 Eigen::Vector3d p_wl;
74 int id;
75
76 // Landmark 1
77 id = 1;
78 p_wl << 0,-1,0; //HACK 这是什么，怎么确定的，静态地标还是动态地标
79 prior_landmarks.insert(pair<int,Eigen::Vector3d> (id, p_wl));
80
81 // // Landmark 2
82 // id = 2;
83 // p_wl << 1,1,-0.5;
84 // prior_landmarks.insert(pair<int,Eigen::Vector3d> (id, p_wl));
85
86 // Landmark 3
87 id = 3;
88 p_wl << 2,-1,0.5;
89 prior_landmarks.insert(pair<int,Eigen::Vector3d> (id, p_wl));
90
91 // Store landmarks for localization
92 filter.setPriorLandmarks(prior_landmarks);
93
94
95 // Open data file
96 ifstream infile("./src/data/imu_landmark_measurements.txt");
97 string line;
98 Eigen::Matrix<double,6,1> imu_measurement = Eigen::Matrix<double,6,1>::Zero();
99 Eigen::Matrix<double,6,1> imu_measurement_prev = Eigen::Matrix<double,6,1>::Zero();
100 double t = 0;
101 double t_prev = 0;
102
103 // Loop through data file and read in measurements line by line
104 while (getline(infile, line)){
105 vector<string> measurement;
106 boost::split(measurement,line,boost::is_any_of(" "));
107 // Handle measurements
108 if (measurement[0].compare("IMU")==0){
109 cout << "Received IMU Data, propagating state\n";
110 assert((measurement.size()-2) == 6);
111 t = stod98(measurement[1]);
112 imu_measurement << stod98(measurement[2]),
113 stod98(measurement[3]),
114 stod98(measurement[4]),
115 stod98(measurement[5]),
116 stod98(measurement[6]),
117 stod98(measurement[7]);
118
119 // Propagate using IMU data
120 double dt = t - t_prev;
121 if (dt > DT_MIN && dt < DT_MAX) {
122 filter.Propagate(imu_measurement_prev, dt);
123 }
124 }
125 // LANDMARK time 地标id 地标相对位移p*3 地标id 地标相对位移p*3
126 else if (measurement[0].compare("LANDMARK")==0){ //地标行
127 cout << "Received LANDMARK observation, correcting state\n";
128 assert((measurement.size()-2)%4 == 0);
129 t = stoi98(measurement[1]);
130 vectorLandmarks measured_landmarks; // 观测到的Landmark 对象列表<id 地标相对body的p.3x1>
131 for (int i=2; i<measurement.size(); i+=4) {
132 int id = stoi98(measurement[i]);
133 Eigen::Vector3d p_bl;
134 p_bl << stoi98(measurement[i+1]),
135 stoi98(measurement[i+2]),
136 stoi98(measurement[i+3]);
137 Landmark landmark(id, p_bl);
138 measured_landmarks.push_back(landmark);
139 }
140
141 // Correct state using landmark measurements
142 filter.CorrectLandmarks(measured_landmarks); //使用地标进行更新
143 }
144
145 // Store previous timestamp
146 t_prev = t;
147 imu_measurement_prev = imu_measurement;
148 }
149
150 // Print final state
151 cout << filter.getState() << endl;
152 return 0;
153}
