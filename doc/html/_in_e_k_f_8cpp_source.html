103void InEKF::setContacts(vector<pair<int,bool> > contacts) {
104#if INEKF_USE_MUTEX
105 lock_guard<mutex> mlock(estimated_contacts_mutex_);
106#endif
107 // Insert new measured contact states
108 for (vector<pair<int,bool> >::iterator it=contacts.begin(); it!=contacts.end(); ++it) {
109 // 将contacts中的(int bool)插入到contacts_中,然后插入结果赋值给ret
110 // 若插入成功, ret.first的迭代器指向contacts_中新插入的(int bool), ret.second==true
111 // 若插入失败, 代表contacts_中已经有相同的key了, ret.first的迭代器指向contacts_中原有的(int bool), ret.second==false
112 pair<map<int,bool>::iterator,bool> ret = contacts_.insert(*it);
113 // 如果插入失败,说明contacts_中已经有这个contacts的key了,则用contacts中的新value替换原有contacts_中的value
114 if (ret.second==false) {
115 ret.first->second = it->second;
116 }
117 }
118 return;

130void InEKF::Propagate(const Eigen::Matrix<double,6,1>& m, double dt) {
131
139 Eigen::Vector3d w = m.head(3) - state_.getGyroscopeBias(); // Angular Velocity
140 Eigen::Vector3d a = m.tail(3) - state_.getAccelerometerBias(); // Linear Acceleration
141
142 Eigen::MatrixXd X = state_.getX();
143 Eigen::MatrixXd P = state_.getP();
144
145 // Extract State
146 Eigen::Matrix3d R = state_.getRotation(); //3x3
147 Eigen::Vector3d v = state_.getVelocity(); //3x1
148 Eigen::Vector3d p = state_.getPosition(); //3x1
149
156 Eigen::Vector3d phi = w*dt;
157 Eigen::Matrix3d R_pred = R * Exp_SO3(phi);
158 Eigen::Vector3d v_pred = v + (R*a + g_)*dt;
159 Eigen::Vector3d p_pred = p + v*dt + 0.5*(R*a + g_)*dt*dt;
160
161 // 设置新状态
162 state_.setRotation(R_pred);
163 state_.setVelocity(v_pred);
164 state_.setPosition(p_pred);
165
166 // ---- Linearized invariant error dynamics -----
176 int dimX = state_.dimX();
177 int dimP = state_.dimP();
178 int dimTheta = state_.dimTheta();
179 Eigen::MatrixXd A = Eigen::MatrixXd::Zero(dimP,dimP);
180 // Inertial terms
181 A.block<3,3>(3,0) = skew(g_); // TODO: Efficiency could be improved by not computing the constant terms every time
182 A.block<3,3>(6,3) = Eigen::Matrix3d::Identity();
183 // Bias terms
184 A.block<3,3>(0,dimP-dimTheta) = -R;
185 A.block<3,3>(3,dimP-dimTheta+3) = -R;
186 for (int i=3; i<dimX; ++i) {
187 A.block<3,3>(3*i-6,dimP-dimTheta) = -skew(X.block<3,1>(0,i))*R;
188 }
189 // cout << A << endl; // XXX:
190
200 Eigen::MatrixXd Qk = Eigen::MatrixXd::Zero(dimP,dimP); // Landmark noise terms will remain zero
201 Qk.block<3,3>(0,0) = noise_params_.getGyroscopeCov();
202 Qk.block<3,3>(3,3) = noise_params_.getAccelerometerCov();
203 //HACK 下面不是很懂，为什么不对estimated_landmarks_操作
204 for(map<int,int>::iterator it=estimated_contact_positions_.begin(); it!=estimated_contact_positions_.end(); ++it) {
205 Qk.block<3,3>(3+3*(it->second-3),3+3*(it->second-3)) = noise_params_.getContactCov(); // Contact noise terms
206 }
207 Qk.block<3,3>(dimP-dimTheta,dimP-dimTheta) = noise_params_.getGyroscopeBiasCov();
208 Qk.block<3,3>(dimP-dimTheta+3,dimP-dimTheta+3) = noise_params_.getAccelerometerBiasCov();
209 // cout << Qk << endl; // XXX:
210
226 Eigen::MatrixXd I = Eigen::MatrixXd::Identity(dimP,dimP);
227 Eigen::MatrixXd Phi = I + A*dt; // TODO: Fast approximation of exp(A*dt). explore using the full exp() instead
228 Eigen::MatrixXd Adj = I;
229 Adj.block(0,0,dimP-dimTheta,dimP-dimTheta) = Adjoint_SEK3(X); // Approx 200 microseconds
230 Eigen::MatrixXd PhiAdj = Phi * Adj;
231 Eigen::MatrixXd Qk_hat = PhiAdj * Qk * PhiAdj.transpose() * dt; // Approximated discretized noise matrix (faster by 400 microseconds)
232
233 // Propagate Covariance
234 Eigen::MatrixXd P_pred = Phi * P * Phi.transpose() + Qk_hat;
235
236 // Set new covariance
237 state_.setP(P_pred);
238
239 return;
240}

252void InEKF::Correct(const Observation &obs)
253{
254 // Compute Kalman Gain
255 Eigen::MatrixXd P = state_.getP();
256 Eigen::MatrixXd PHT = P * obs.H.transpose();
257 Eigen::MatrixXd S = obs.H * PHT + obs.N;
258 Eigen::MatrixXd K = PHT * S.inverse();
259
260 // Copy X along the diagonals if more than one measurement
261 Eigen::MatrixXd BigX;
262 state_.copyDiagX(obs.Y.rows() / state_.dimX(), BigX);
263
264 // Compute correction terms
265 Eigen::MatrixXd Z = BigX * obs.Y - obs.b;
266 Eigen::VectorXd delta = K * obs.PI * Z;
267 Eigen::MatrixXd dX = Exp_SEK3(delta.segment(0, delta.rows() - state_.dimTheta()));
268 Eigen::VectorXd dTheta = delta.segment(delta.rows() - state_.dimTheta(), state_.dimTheta());
269
270 // Update state
271 Eigen::MatrixXd X_new = dX * state_.getX(); // Right-Invariant Update
272 Eigen::VectorXd Theta_new = state_.getTheta() + dTheta;
273 state_.setX(X_new);
274 state_.setTheta(Theta_new);
275
276 // Update Covariance
277 Eigen::MatrixXd IKH = Eigen::MatrixXd::Identity(state_.dimP(), state_.dimP()) - K * obs.H;
278 Eigen::MatrixXd P_new = IKH * P * IKH.transpose() + K * obs.N * K.transpose(); // Joseph update form
279
280 state_.setP(P_new);
281}

285void InEKF::CorrectLandmarks(const vectorLandmarks &measured_landmarks)
286{
287#if INEKF_USE_MUTEX
288 lock_guard<mutex> mlock(estimated_landmarks_mutex_);
289#endif
290 Eigen::VectorXd Y;
291 Eigen::VectorXd b;
292 Eigen::MatrixXd H;
293 Eigen::MatrixXd N;
294 Eigen::MatrixXd PI;
295
296 Eigen::Matrix3d R = state_.getRotation();
297 vectorLandmarks new_landmarks; // 新增的landmarks列表
298 vector<int> used_landmark_ids; // 储存处理过的landmark id,避免一帧数据有相同id的lm
299
301 for (vectorLandmarksIterator it = measured_landmarks.begin(); it != measured_landmarks.end(); ++it)
302 {
304 if (find(used_landmark_ids.begin(), used_landmark_ids.end(), it->id) != used_landmark_ids.end())
305 {
306 cout << "Duplicate landmark ID detected! Skipping measurement.\n";
307 continue;
308 }
309 else
310 {
311 used_landmark_ids.push_back(it->id); // 将id添加到处理过的列表中,下面对此id的观测数据进行处理
312 }
313
315 mapIntVector3dIterator it_prior = prior_landmarks_.find(it->id);
316 map<int, int>::iterator it_estimated = estimated_landmarks_.find(it->id);
318 if (it_prior != prior_landmarks_.end())
319 {
320 // Found in prior landmark set
321 int dimX = state_.dimX();
322 int dimP = state_.dimP();
323 int startIndex;
324
332 startIndex = Y.rows();
333 Y.conservativeResize(startIndex + dimX, Eigen::NoChange); // Y矩阵新增dimX行,不改变列数
334 Y.segment(startIndex, dimX) = Eigen::VectorXd::Zero(dimX); //初始化为0
335 Y.segment(startIndex, 3) = it->position; // p_bl
336 Y(startIndex + 4) = 1;
337
345 startIndex = b.rows();
346 b.conservativeResize(startIndex + dimX, Eigen::NoChange);
347 b.segment(startIndex, dimX) = Eigen::VectorXd::Zero(dimX);
348 b.segment(startIndex, 3) = it_prior->second; // p_wl
349 b(startIndex + 4) = 1;
350
358 startIndex = H.rows();
359 H.conservativeResize(startIndex + 3, dimP);
360 H.block(startIndex, 0, 3, dimP) = Eigen::MatrixXd::Zero(3, dimP);
361 H.block(startIndex, 0, 3, 3) = skew(it_prior->second); // skew(p_wl)
362 H.block(startIndex, 6, 3, 3) = -Eigen::Matrix3d::Identity(); // -I
363
368 startIndex = N.rows();
369 N.conservativeResize(startIndex + 3, startIndex + 3);
370 N.block(startIndex, 0, 3, startIndex) = Eigen::MatrixXd::Zero(3, startIndex); // 下三角置0
371 N.block(0, startIndex, startIndex, 3) = Eigen::MatrixXd::Zero(startIndex, 3); // 上三角置0
372 N.block(startIndex, startIndex, 3, 3) = R * noise_params_.getLandmarkCov() * R.transpose();
373
378 startIndex = PI.rows();
379 int startIndex2 = PI.cols();
380 PI.conservativeResize(startIndex + 3, startIndex2 + dimX);
381 PI.block(startIndex, 0, 3, startIndex2) = Eigen::MatrixXd::Zero(3, startIndex2); // 从第二个观测起生效,下三角置0
382 PI.block(0, startIndex2, startIndex, dimX) = Eigen::MatrixXd::Zero(startIndex, dimX); // 从第二个观测起生效,上三角置0
383 PI.block(startIndex, startIndex2, 3, dimX) = Eigen::MatrixXd::Zero(3, dimX); // 对角位置的新PI全部置0
384 PI.block(startIndex, startIndex2, 3, 3) = Eigen::Matrix3d::Identity(); // 新的PI前3x3为单位阵
385 }
387 else if (it_estimated != estimated_landmarks_.end())
388 {
389 // Found in estimated landmark set
390 int dimX = state_.dimX();
391 int dimP = state_.dimP();
392 int startIndex;
393
406 startIndex = Y.rows();
407 Y.conservativeResize(startIndex + dimX, Eigen::NoChange);
408 Y.segment(startIndex, dimX) = Eigen::VectorXd::Zero(dimX);
409 Y.segment(startIndex, 3) = it->position; // p_bl
410 Y(startIndex + 4) = 1;
411 Y(startIndex + it_estimated->second) = -1;
412
425 startIndex = b.rows();
426 b.conservativeResize(startIndex + dimX, Eigen::NoChange);
427 b.segment(startIndex, dimX) = Eigen::VectorXd::Zero(dimX);
428 b(startIndex + 4) = 1;
429 b(startIndex + it_estimated->second) = -1;
430
439 startIndex = H.rows();
440 H.conservativeResize(startIndex + 3, dimP);
441 H.block(startIndex, 0, 3, dimP) = Eigen::MatrixXd::Zero(3, dimP);
442 H.block(startIndex, 6, 3, 3) = -Eigen::Matrix3d::Identity(); // -I
443 H.block(startIndex, 3 * it_estimated->second - 6, 3, 3) = Eigen::Matrix3d::Identity(); // I
444
447 startIndex = N.rows();
448 N.conservativeResize(startIndex + 3, startIndex + 3);
449 N.block(startIndex, 0, 3, startIndex) = Eigen::MatrixXd::Zero(3, startIndex);
450 N.block(0, startIndex, startIndex, 3) = Eigen::MatrixXd::Zero(startIndex, 3);
451 N.block(startIndex, startIndex, 3, 3) = R * noise_params_.getLandmarkCov() * R.transpose();
452
460 startIndex = PI.rows();
461 int startIndex2 = PI.cols();
462 PI.conservativeResize(startIndex + 3, startIndex2 + dimX);
463 PI.block(startIndex, 0, 3, startIndex2) = Eigen::MatrixXd::Zero(3, startIndex2);
464 PI.block(0, startIndex2, startIndex, dimX) = Eigen::MatrixXd::Zero(startIndex, dimX);
465 PI.block(startIndex, startIndex2, 3, dimX) = Eigen::MatrixXd::Zero(3, dimX);
466 PI.block(startIndex, startIndex2, 3, 3) = Eigen::Matrix3d::Identity();
467 }
469 else
470 {
471 new_landmarks.push_back(*it);
472 }
473 }
474
477 Observation obs(Y, b, H, N, PI);
478 if (!obs.empty())
479 {
480 this->Correct(obs);
481 }
482
485 if (new_landmarks.size() > 0)
486 {
487 Eigen::MatrixXd X_aug = state_.getX();
488 Eigen::MatrixXd P_aug = state_.getP();
489 Eigen::Vector3d p = state_.getPosition();
490 // 循环对每个新增动态地标操作
491 for (vectorLandmarksIterator it = new_landmarks.begin(); it != new_landmarks.end(); ++it)
492 {
513 int startIndex = X_aug.rows();
514 X_aug.conservativeResize(startIndex + 1, startIndex + 1); // 大小加1
515 X_aug.block(startIndex, 0, 1, startIndex) = Eigen::MatrixXd::Zero(1, startIndex); //下三角置0
516 X_aug.block(0, startIndex, startIndex, 1) = Eigen::MatrixXd::Zero(startIndex, 1); //上三角置0
517 X_aug(startIndex, startIndex) = 1; //新增对角置1
518 X_aug.block(0, startIndex, 3, 1) = p + R * it->position;
519
562 Eigen::MatrixXd F = Eigen::MatrixXd::Zero(state_.dimP() + 3, state_.dimP()); // F阵
563 F.block(0, 0, state_.dimP() - state_.dimTheta(), state_.dimP() - state_.dimTheta()) = Eigen::MatrixXd::Identity(state_.dimP() - state_.dimTheta(), state_.dimP() - state_.dimTheta()); // for old X
564 F.block(state_.dimP() - state_.dimTheta(), 6, 3, 3) = Eigen::Matrix3d::Identity(); // for new landmark
565 F.block(state_.dimP() - state_.dimTheta() + 3, state_.dimP() - state_.dimTheta(), state_.dimTheta(), state_.dimTheta()) = Eigen::MatrixXd::Identity(state_.dimTheta(), state_.dimTheta()); // for theta
566 Eigen::MatrixXd G = Eigen::MatrixXd::Zero(F.rows(), 3); // G阵
567 G.block(G.rows() - state_.dimTheta() - 3, 0, 3, 3) = R;
568 P_aug = (F * P_aug * F.transpose() + G * noise_params_.getLandmarkCov() * G.transpose()).eval(); //.eval的作用是让Eigen显示计算链式矩阵运算，并立刻返回结果，避免大型链式计算中可能存在的错误发生
569
570 // 更新状态和协方差
571 state_.setX(X_aug);
572 state_.setP(P_aug);
573
575 estimated_landmarks_.insert(pair<int, int>(it->id, startIndex));
576 }
577 }
578 return;
579}

584void InEKF::CorrectKinematics(const vectorKinematics &measured_kinematics)
585{
586#if INEKF_USE_MUTEX
587 lock_guard<mutex> mlock(estimated_contacts_mutex_);
588#endif
589 Eigen::VectorXd Y;
590 Eigen::VectorXd b;
591 Eigen::MatrixXd H;
592 Eigen::MatrixXd N;
593 Eigen::MatrixXd PI;
594
595 Eigen::Matrix3d R = state_.getRotation();
596 vector<pair<int, int>> remove_contacts; // 要删除的腿列表,<id,id在X中的索引>
597 vectorKinematics new_contacts; // 要新增的腿列表
598 vector<int> used_contact_ids; // 已处理过的contact id列表
599
601 for (vectorKinematicsIterator it = measured_kinematics.begin(); it != measured_kinematics.end(); ++it)
602 {
604 if (find(used_contact_ids.begin(), used_contact_ids.end(), it->id) != used_contact_ids.end())
605 {
606 cout << "Duplicate contact ID detected! Skipping measurement.\n";
607 continue;
608 }
609 else
610 {
611 used_contact_ids.push_back(it->id); // 记录已使用的id(下面将要处理该id)
612 }
613
615 map<int, bool>::iterator it_contact = contacts_.find(it->id);
616 if (it_contact == contacts_.end())
617 {
618 continue;
619 }
620 bool contact_indicated = it_contact->second;
621
623 map<int, int>::iterator it_estimated = estimated_contact_positions_.find(it->id); // 该id在estimated_contact_positions_中的迭代器
624 bool found = it_estimated != estimated_contact_positions_.end();
625
627 if (!contact_indicated && found)
628 {
629 remove_contacts.push_back(*it_estimated);
630 }
632 else if (contact_indicated && !found)
633 {
634 new_contacts.push_back(*it);
635 }
637 else if (contact_indicated && found)
638 {
639 int dimX = state_.dimX();
640 int dimP = state_.dimP();
641 int startIndex;
642
655 startIndex = Y.rows(); // 开始索引,多个观测时,Y中已经有数据了
656 Y.conservativeResize(startIndex + dimX, Eigen::NoChange); // Y矩阵新增dimX行,不改变列数
657 Y.segment(startIndex, dimX) = Eigen::VectorXd::Zero(dimX);
658 Y.segment(startIndex, 3) = it->pose.block<3, 1>(0, 3); // p_bc
659 Y(startIndex + 4) = 1;
660 Y(startIndex + it_estimated->second) = -1; // it_estimated代表当前估计的腿在状态X中的列位置
661
674 startIndex = b.rows();
675 b.conservativeResize(startIndex + dimX, Eigen::NoChange);
676 b.segment(startIndex, dimX) = Eigen::VectorXd::Zero(dimX);
677 b(startIndex + 4) = 1;
678 b(startIndex + it_estimated->second) = -1;
679
688 startIndex = H.rows();
689 H.conservativeResize(startIndex + 3, dimP);
690 H.block(startIndex, 0, 3, dimP) = Eigen::MatrixXd::Zero(3, dimP);
691 H.block(startIndex, 6, 3, 3) = -Eigen::Matrix3d::Identity(); // -I
692 H.block(startIndex, 3 * it_estimated->second - 6, 3, 3) = Eigen::Matrix3d::Identity(); // I
693
696 startIndex = N.rows();
697 N.conservativeResize(startIndex + 3, startIndex + 3);
698 N.block(startIndex, 0, 3, startIndex) = Eigen::MatrixXd::Zero(3, startIndex); //从第二个观测起生效,下三角置0
699 N.block(0, startIndex, startIndex, 3) = Eigen::MatrixXd::Zero(startIndex, 3); //从第二个观测起生效,上三角置0
700 N.block(startIndex, startIndex, 3, 3) = R * it->covariance.block<3, 3>(3, 3) * R.transpose(); //对角放新的3X3的N矩阵
701
709 startIndex = PI.rows();
710 int startIndex2 = PI.cols();
711 PI.conservativeResize(startIndex + 3, startIndex2 + dimX);
712 PI.block(startIndex, 0, 3, startIndex2) = Eigen::MatrixXd::Zero(3, startIndex2); //从第二个观测起生效,下三角置0
713 PI.block(0, startIndex2, startIndex, dimX) = Eigen::MatrixXd::Zero(startIndex, dimX); //从第二个观测起生效,上三角置0
714 PI.block(startIndex, startIndex2, 3, dimX) = Eigen::MatrixXd::Zero(3, dimX); //对角放新的PI全部置0
715 PI.block(startIndex, startIndex2, 3, 3) = Eigen::Matrix3d::Identity(); //新的PI前3x3为单位阵
716 }
718 else
719 {
720 continue;
721 }
722 }
723
725 Observation obs(Y, b, H, N, PI);
726 if (!obs.empty())
727 {
728 this->Correct(obs);
729 }
730
732 if (remove_contacts.size() > 0)
733 {
734 Eigen::MatrixXd X_rem = state_.getX();
735 Eigen::MatrixXd P_rem = state_.getP();
736 for (vector<pair<int, int>>::iterator it = remove_contacts.begin(); it != remove_contacts.end(); ++it)
737 {
739 estimated_contact_positions_.erase(it->first);
740
742 removeRowAndColumn(X_rem, it->second);
743
745 int startIndex = 3 + 3 * (it->second - 3);
746 removeRowAndColumn(P_rem, startIndex); // TODO: Make more efficient
747 removeRowAndColumn(P_rem, startIndex); // TODO: Make more efficient
748 removeRowAndColumn(P_rem, startIndex); // TODO: Make more efficient
749
751 for (map<int, int>::iterator it2 = estimated_landmarks_.begin(); it2 != estimated_landmarks_.end(); ++it2)
752 {
753 if (it2->second > it->second)
754 it2->second -= 1;
755 }
756 for (map<int, int>::iterator it2 = estimated_contact_positions_.begin(); it2 != estimated_contact_positions_.end(); ++it2)
757 {
758 if (it2->second > it->second)
759 it2->second -= 1;
760 }
762 for (vector<pair<int, int>>::iterator it2 = it; it2 != remove_contacts.end(); ++it2)
763 {
764 if (it2->second > it->second)
765 it2->second -= 1;
766 }
767
768 state_.setX(X_rem);
769 state_.setP(P_rem);
770 }
771 }
772
774 if (new_contacts.size() > 0)
775 {
776 Eigen::MatrixXd X_aug = state_.getX();
777 Eigen::MatrixXd P_aug = state_.getP();
778 Eigen::Vector3d p = state_.getPosition();
779 // 循环对每个新增腿进行操作
780 for (vectorKinematicsIterator it = new_contacts.begin(); it != new_contacts.end(); ++it)
781 {
802 int startIndex = X_aug.rows();
803 X_aug.conservativeResize(startIndex + 1, startIndex + 1); //将X大小扩充1
804 X_aug.block(startIndex, 0, 1, startIndex) = Eigen::MatrixXd::Zero(1, startIndex); //新增下三角部分置0
805 X_aug.block(0, startIndex, startIndex, 1) = Eigen::MatrixXd::Zero(startIndex, 1); //新增上三角部分置0
806 X_aug(startIndex, startIndex) = 1; //新增对角线部分置1
807 X_aug.block(0, startIndex, 3, 1) = p + R * it->pose.block<3, 1>(0, 3); // 更新腿的状态
808
851 Eigen::MatrixXd F = Eigen::MatrixXd::Zero(state_.dimP() + 3, state_.dimP()); // F阵
852 F.block(0, 0, state_.dimP() - state_.dimTheta(), state_.dimP() - state_.dimTheta()) = Eigen::MatrixXd::Identity(state_.dimP() - state_.dimTheta(), state_.dimP() - state_.dimTheta()); // for old X
853 F.block(state_.dimP() - state_.dimTheta(), 6, 3, 3) = Eigen::Matrix3d::Identity(); // for new landmark
854 F.block(state_.dimP() - state_.dimTheta() + 3, state_.dimP() - state_.dimTheta(), state_.dimTheta(), state_.dimTheta()) = Eigen::MatrixXd::Identity(state_.dimTheta(), state_.dimTheta()); // for theta
855 Eigen::MatrixXd G = Eigen::MatrixXd::Zero(F.rows(), 3); // G阵
856 G.block(G.rows() - state_.dimTheta() - 3, 0, 3, 3) = R;
857 P_aug = (F * P_aug * F.transpose() + G * it->covariance.block<3, 3>(3, 3) * G.transpose()).eval(); //.eval的作用是让Eigen显示计算链式矩阵运算，并立刻返回结果，避免大型链式计算中可能存在的错误发生
858
859 // Update state and covariance
860 state_.setX(X_aug);
861 state_.setP(P_aug);
862
864 estimated_contact_positions_.insert(pair<int, int>(it->id, startIndex));
865 }
866 }
867
868 return;
869}

872void removeRowAndColumn(Eigen::MatrixXd& M, int index) {
873 unsigned int dimX = M.cols();
874 // cout << "Removing index: " << index<< endl;
875 // 把矩阵index行下面的dim-index-1行用最后dim-index-1行代替
876 M.block(index,0,dimX-index-1,dimX) = M.bottomRows(dimX-index-1).eval();
877 // 把矩阵index列右边的dim-index-1列用最右边的dim-index-1列代替
878 M.block(0,index,dimX,dimX-index-1) = M.rightCols(dimX-index-1).eval();
879 // 然后把矩阵重新resize
880 M.conservativeResize(dimX-1,dimX-1);
881}
